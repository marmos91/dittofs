    // server from TestMain or a `setupCrossProtocol(t)` helper, call the same one.
    fixture := setupCrossProtocolFixture(t)   // existing pattern; reuse do not recreate
    defer fixture.Cleanup()

    // Table-driven over the ~18 triggerable codes. Each entry pairs a trigger
    // with (wantNFSErrno, wantSMBStatus). The want values come directly from
    // common/'s errorMap for this code — either imported or hand-copied if
    // test/e2e cannot import internal/ packages.
    cases := []struct {
        name          string
        triggerNFS    func(t *testing.T, mountRoot string) syscall.Errno
        triggerSMB    func(t *testing.T, mountRoot string) uint32   // NT status
        wantNFSErrno  syscall.Errno
        wantSMBStatus uint32
    }{
        {
            name:          "ErrNotFound",
            triggerNFS:    helpers.TriggerErrNotFound_NFS,
            triggerSMB:    helpers.TriggerErrNotFound_SMB,
            wantNFSErrno:  syscall.ENOENT,      // MapToNFS3(ErrNotFound) -> NFS3ErrNoEnt -> ENOENT at kernel
            wantSMBStatus: 0xC0000034,          // StatusObjectNameNotFound
        },
        {
            name:          "ErrAlreadyExists",
            triggerNFS:    helpers.TriggerErrAlreadyExists_NFS,
            triggerSMB:    helpers.TriggerErrAlreadyExists_SMB,
            wantNFSErrno:  syscall.EEXIST,
            wantSMBStatus: 0xC0000035,          // StatusObjectNameCollision
        },
        // ... 16 more cases per D-13 e2e-triggerable list ...
    }

    for _, c := range cases {
        t.Run(c.name, func(t *testing.T) {
            gotNFS := c.triggerNFS(t, fixture.NFSMount)
            if gotNFS != c.wantNFSErrno {
                t.Errorf("NFS %s: got %v, want %v", c.name, gotNFS, c.wantNFSErrno)
            }

            gotSMB := c.triggerSMB(t, fixture.SMBMount)
            if gotSMB != c.wantSMBStatus {
                t.Errorf("SMB %s: got 0x%08x, want 0x%08x", c.name, gotSMB, c.wantSMBStatus)
            }
        })
    }
}

Add an "Error mapping" section + pool notes:

```
## Error mapping

`metadata.ErrorCode` values are translated to SMB `NTSTATUS` codes by
`internal/adapter/common.MapToSMB`, which reads the same consolidated
table as NFSv3 and NFSv4 (`internal/adapter/common/errmap.go`). Lock-
operation context uses `common.MapLockToSMB`, which overrides a few
codes (e.g., `ErrLocked` -> `STATUS_LOCK_NOT_GRANTED` in lock context vs.
`STATUS_FILE_LOCK_CONFLICT` in general I/O context).

The consolidation uses `errors.As` for StoreError unwrapping, which is
the standard Go pattern; prior to v0.15.0 the SMB handler used an
unwrapped type assertion that failed on wrapped errors.

## READ response buffers

SMB2 READ responses allocate their data buffer through
`internal/adapter/pool` (4 KB / 64 KB / 1 MB tiers with direct-alloc
fallback for larger sizes). The pool buffer is returned via the
`SMBResponseBase.ReleaseData` closure, which the response encoder
invokes after `WriteNetBIOSFrame` completes — safe across plain,
encrypted, and compound-response paths.
```

## Adding a new metadata.ErrorCode

To add a new `metadata.ErrorCode` that is visible to NFS and SMB clients:

1. Add the constant to `pkg/metadata/errors/errors.go` and re-export it

```
merrs.ErrFooBar: {
    NFS3: nfs3types.NFS3Err...,
    NFS4: nfs4types.NFS4ERR_...,
    SMB:  smbtypes.Status...,
},
```

   The struct literal requires all three columns — the type system
   enforces that you do not forget a protocol.

3. If the code is lock-operation-specific, add a row in
   `internal/adapter/common/lock_errmap.go`'s `lockErrorMap` with the
   lock-context codes.
