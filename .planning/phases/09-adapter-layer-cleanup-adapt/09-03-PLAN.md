package common

import (
    goerrors "errors"

    nfs3types "github.com/marmos91/dittofs/internal/adapter/nfs/types"
    nfs4types "github.com/marmos91/dittofs/internal/adapter/nfs/v4/types"
    smbtypes "github.com/marmos91/dittofs/internal/adapter/smb/types"
    merrs "github.com/marmos91/dittofs/pkg/metadata/errors"
)

// protoCodes carries the per-protocol code for a single merrs.ErrorCode.
// Every row in errorMap populates all three columns — the type system
// guarantees that adding a new ErrorCode adds all three codes at once
// (ADAPT-03 "one edit, not two" contract).
type protoCodes struct {
    NFS3 uint32
    NFS4 uint32
    SMB  smbtypes.Status
}

// defaultCodes are returned when the error is not a *merrs.StoreError OR when
// the StoreError.Code is not in errorMap. Matches existing behavior of the
// consolidated protocol-specific switches (which each had their own default).
var defaultCodes = protoCodes{
    NFS3: nfs3types.NFS3ErrIO,
    NFS4: nfs4types.NFS4ERR_SERVERFAULT,
    SMB:  smbtypes.StatusInternalError,
}

// errorMap is the single source of truth for merrs.ErrorCode -> protocol code.
//
// Authority per column (derived during ADAPT-03 consolidation):
//   - NFS3 column: internal/adapter/nfs/xdr/errors.go (MapStoreErrorToNFSStatus
//     is more complete than create.go's in-handler switch per D-07 gotcha #2
//     in PATTERNS.md; in particular, ErrLocked -> NFS3ErrJukebox comes from
//     xdr/errors.go:140-146).
//   - NFS4 column: internal/adapter/nfs/v4/types/errors.go:11-70.
//   - SMB column: internal/adapter/smb/v2/handlers/converters.go:354-395 for
//     general context; lock-context deltas go in lock_errmap.go instead.
//
// Codes the existing per-protocol switches OMITTED (planning-time findings per
// D-07) are populated here using the most conservative fallback for that
// protocol — e.g. SMB has no EPERM distinction, so ErrPermissionDenied -> StatusAccessDenied.
var errorMap = map[merrs.ErrorCode]protoCodes{
    merrs.ErrNotFound: {
        NFS3: nfs3types.NFS3ErrNoEnt,
        NFS4: nfs4types.NFS4ERR_NOENT,
        SMB:  smbtypes.StatusObjectNameNotFound,
    },
    merrs.ErrAccessDenied: {
        NFS3: nfs3types.NFS3ErrAccess,
        NFS4: nfs4types.NFS4ERR_ACCESS,
        SMB:  smbtypes.StatusAccessDenied,
    },
    // ... complete the table. Every row MUST populate all three columns.
    // Executor: transcribe from the three source switches during read-first.
}

// MapToNFS3 translates err to an NFS3 status. Returns NFS3OK for nil,
// defaultCodes.NFS3 if err is not a *merrs.StoreError, and the table value
// otherwise. Uses goerrors.As so wrapped StoreErrors are unwrapped correctly.
func MapToNFS3(err error) uint32 {
    if err == nil {
        return nfs3types.NFS3OK
    }
    var storeErr *merrs.StoreError
    if !goerrors.As(err, &storeErr) {
        return defaultCodes.NFS3
    }
    if codes, ok := errorMap[storeErr.Code]; ok {
        return codes.NFS3
    }
    return defaultCodes.NFS3
}

// MapToNFS4 and MapToSMB follow the identical structure (use goerrors.As + merrs.StoreError).
func MapToNFS4(err error) uint32 { /* ... */ }
func MapToSMB(err error) smbtypes.Status { /* ... */ }

2c. `internal/adapter/common/lock_errmap.go` — per D-08 §3:

Lock-operation-specific table. `merrs.ErrLocked` maps differently in lock context vs general context (see PATTERNS.md). The lock table is keyed on the same `merrs.ErrorCode` but returns lock-operation codes.

```
package common

import (
    goerrors "errors"

    nfs3types "github.com/marmos91/dittofs/internal/adapter/nfs/types"
    nfs4types "github.com/marmos91/dittofs/internal/adapter/nfs/v4/types"
    smbtypes "github.com/marmos91/dittofs/internal/adapter/smb/types"
    merrs "github.com/marmos91/dittofs/pkg/metadata/errors"
)

// lockErrorMap mirrors errorMap but with lock-operation-specific protocol codes.
// Context matters: in a LOCK request (SMB2 LOCK, NLM, NFS4 LOCK/LOCKU) ErrLocked
// means "denied, try again" (STATUS_LOCK_NOT_GRANTED / NFS3ErrJukebox /
// NFS4ERR_DENIED). In the general I/O path ErrLocked means
// STATUS_FILE_LOCK_CONFLICT (see errorMap).
//
// Source: internal/adapter/smb/v2/handlers/lock.go:533-549 (SMB column);
// internal/adapter/nfs/v4/types/errors.go:59-64 (NFS4 column);
// internal/adapter/nfs/xdr/errors.go:140-146 (NFS3 column).
var lockErrorMap = map[merrs.ErrorCode]protoCodes{
    merrs.ErrLocked: {
        NFS3: nfs3types.NFS3ErrJukebox,
        NFS4: nfs4types.NFS4ERR_DENIED,
        SMB:  smbtypes.StatusLockNotGranted,
    },
    merrs.ErrLockNotFound: {
        NFS3: nfs3types.NFS3ErrInval,
        NFS4: nfs4types.NFS4ERR_LOCK_RANGE,
        SMB:  smbtypes.StatusRangeNotLocked,
    },
    merrs.ErrLockConflict:       { /* ... */ },
    merrs.ErrDeadlock:           { NFS4: nfs4types.NFS4ERR_DEADLOCK, SMB: smbtypes.StatusLockNotGranted /* SMB no direct code */, /* NFS3: ... */ },
    merrs.ErrGracePeriod:        { NFS4: nfs4types.NFS4ERR_GRACE, NFS3: nfs3types.NFS3ErrJukebox /* NFSv3 no direct */, SMB: smbtypes.StatusInternalError },
    merrs.ErrLockLimitExceeded:  { /* ... */ },
    // Also include the lock-adjacent errors that lockErrorToStatus handles today
    // (ErrNotFound -> StatusFileClosed, ErrPermissionDenied -> StatusAccessDenied,
    // ErrIsDirectory -> StatusFileIsADirectory). These are lock-context overrides.
    merrs.ErrNotFound:           { SMB: smbtypes.StatusFileClosed, NFS3: ..., NFS4: ... },
    merrs.ErrPermissionDenied:   { SMB: smbtypes.StatusAccessDenied, NFS3: ..., NFS4: ... },
    merrs.ErrIsDirectory:        { SMB: smbtypes.StatusFileIsADirectory, NFS3: ..., NFS4: ... },
}

// MapLockToSMB — fallback chain: lockErrorMap, then errorMap (general), then defaults.
func MapLockToSMB(err error) smbtypes.Status {
    if err == nil {
        return smbtypes.StatusSuccess
    }
    var storeErr *merrs.StoreError
    if !goerrors.As(err, &storeErr) {
        return defaultCodes.SMB
    }
    if codes, ok := lockErrorMap[storeErr.Code]; ok {
        return codes.SMB
    }
    if codes, ok := errorMap[storeErr.Code]; ok {
        return codes.SMB
    }
    return defaultCodes.SMB
}

// MapLockToNFS3 / MapLockToNFS4 follow the same structure.
```
