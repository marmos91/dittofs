<objective>
Surface the new GC engine through `dfsctl` (CLI) and the existing REST API. Operators must be able to trigger a GC run on demand, run it as a dry-run, and inspect the last-run summary. Both surfaces share a single Runtime entry point so behavior is identical between CLI and HTTP.

Purpose: Without a CLI/REST surface, the operator has no way to validate the new GC algorithm in production except by waiting for the periodic timer. Dry-run is essential for first-time deployment confidence (D-09).
Output: Two new dfsctl subcommands, two new REST handlers, the apiclient method, and unit tests for the handlers.
</objective>

       ```go
       type BlockStoreGCRequest struct {

       type BlockStoreGCResponse struct {

       func (h *BlockStoreHandler) RunGC(w http.ResponseWriter, r *http.Request) {
           name := chi.URLParam(r, "name")
           if name == "" {
               BadRequest(w, "share name required")
               return
           }
           var req BlockStoreGCRequest
           if r.ContentLength > 0 {
               if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
                   BadRequest(w, fmt.Sprintf("decode body: %v", err))
                   return
               }
           }
           stats, err := h.runtime.RunBlockGC(r.Context(), name, req.DryRun)
           if err != nil {
               // Map metadata.ExportError values per CLAUDE.md rule 6.
               if errors.Is(err, metadata.ErrNoEntity) { NotFound(w, err.Error()); return }
               InternalError(w, err)
               return
           }
           writeJSON(w, http.StatusOK, BlockStoreGCResponse{Stats: stats})
       }

       func (h *BlockStoreHandler) GCStatus(w http.ResponseWriter, r *http.Request) {
           name := chi.URLParam(r, "name")
           if name == "" {
               BadRequest(w, "share name required")
               return
           }
           // Look up the share's local store dir via Runtime; load last-run.json.
           localDir, err := h.runtime.GetLocalStoreDir(r.Context(), name)
           if err != nil {
               if errors.Is(err, metadata.ErrNoEntity) { NotFound(w, err.Error()); return }
               InternalError(w, err)
               return
           }
           summaryPath := filepath.Join(localDir, "gc-state", "last-run.json")
           data, err := os.ReadFile(summaryPath)
           if err != nil {
               if os.IsNotExist(err) {
                   NotFound(w, "no GC run recorded for share "+name)
                   return
               }
               InternalError(w, err)
               return
           }
           var summary engine.GCRunSummary
           if err := json.Unmarshal(data, &summary); err != nil {
               InternalError(w, fmt.Errorf("parse last-run.json: %w", err))
               return
           }
           writeJSON(w, http.StatusOK, summary)
       }
       ```
