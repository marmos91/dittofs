       func (bc *FSStore) flushBlock(ctx context.Context, payloadID string, blockIdx uint64, mb *memBlock) (string, int64, error) {
           // STAGE under lock — capture immutable snapshot.
           mb.mu.Lock()
           if !mb.dirty || mb.data == nil {
               mb.mu.Unlock()
               return "", 0, nil
           }
           dataSize := mb.dataSize
           staged := bytes.Clone(mb.data[:dataSize])  // O(dataSize) copy — single allocation
           targetPath := bc.blockPath(payloadID, blockIdx)
           mb.mu.Unlock()

           // DISK I/O outside the lock.
           if err := os.MkdirAll(filepath.Dir(targetPath), 0o755); err != nil {
               return "", 0, fmt.Errorf("mkdir for %s: %w", targetPath, err)
           }
           // Write to a .tmp file then rename for atomicity (matches existing pattern).
           tmpPath := targetPath + ".tmp"
           f, err := os.OpenFile(tmpPath, os.O_CREATE|os.O_WRONLY|os.O_TRUNC, 0o644)
           if err != nil {
               return "", 0, fmt.Errorf("open %s: %w", tmpPath, err)
           }
           if _, err := f.Write(staged); err != nil {
               _ = f.Close()
               _ = os.Remove(tmpPath)
               return "", 0, fmt.Errorf("write %s: %w", tmpPath, err)
           }
           if err := f.Sync(); err != nil {
               _ = f.Close()
               _ = os.Remove(tmpPath)
               return "", 0, fmt.Errorf("fsync %s: %w", tmpPath, err)
           }
           if err := f.Close(); err != nil {
               _ = os.Remove(tmpPath)
               return "", 0, fmt.Errorf("close %s: %w", tmpPath, err)
           }
           if err := os.Rename(tmpPath, targetPath); err != nil {
               _ = os.Remove(tmpPath)
               return "", 0, fmt.Errorf("rename %s: %w", targetPath, err)
           }

           // POST-WRITE: re-acquire briefly to flip flags + release the in-memory buffer.
           mb.mu.Lock()
           mb.data = nil
           mb.dataSize = 0
           mb.dirty = false
           mb.mu.Unlock()

         type lruEntry struct {
             hash blockstore.ContentHash
             size int64
             path string  // "blocks/{hh}/{hh}/{hex}"
         }

         func (bc *FSStore) lruTouch(h blockstore.ContentHash, size int64, path string) {
             bc.lruMu.Lock(); defer bc.lruMu.Unlock()
             if el, ok := bc.lruIndex[h]; ok {
                 bc.lruList.MoveToFront(el)
                 return
             }
             el := bc.lruList.PushFront(&lruEntry{hash: h, size: size, path: path})
             bc.lruIndex[h] = el
         }

         func (bc *FSStore) lruEvictOne() (int64, error) {
             bc.lruMu.Lock(); defer bc.lruMu.Unlock()
             el := bc.lruList.Back()
             if el == nil { return 0, errLRUEmpty }
             entry := el.Value.(*lruEntry)
             bc.lruList.Remove(el)
             delete(bc.lruIndex, entry.hash)
             if err := os.Remove(entry.path); err != nil && !os.IsNotExist(err) {
                 return 0, fmt.Errorf("evict %s: %w", entry.path, err)
             }
             return entry.size, nil
         }
