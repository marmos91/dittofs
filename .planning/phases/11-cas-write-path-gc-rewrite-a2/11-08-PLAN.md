       ```go
       // TestExternalVerifier_ContentHashHeader confirms BSCAS-06: every cas/.../
       // object stamped by Phase 11's syncer carries x-amz-meta-content-hash so
       // external tooling (aws s3api head-object) can verify integrity without
       // DittoFS metadata. See D-33.
       func TestExternalVerifier_ContentHashHeader(t *testing.T) {
           if testing.Short() { t.Skip() }
           env := setupS3E2E(t) // existing helper; or use whatever convention test/e2e/ uses
           defer env.Teardown()

           filePath := env.WriteFileViaNFS("hello.bin", randomBytes(4 * 1024 * 1024))
           env.SyncDrain(t)

           // List cas/.../ keys directly via S3 SDK.
           keys := env.ListS3Prefix(t, "cas/")
           require.NotEmpty(t, keys, "expected at least one CAS object after sync")

           for _, key := range keys {
               head, err := env.S3.HeadObject(env.Ctx, &s3.HeadObjectInput{
                   Bucket: aws.String(env.Bucket),
                   Key:    aws.String(key),
               })
               require.NoError(t, err)

               got, ok := head.Metadata["content-hash"]
               require.True(t, ok, "key %s missing content-hash metadata", key)
               require.Truef(t, strings.HasPrefix(got, "blake3:"), "key %s metadata value %q lacks blake3: prefix", key, got)

               // Cross-check by fetching the body and computing BLAKE3 ourselves.
               obj, err := env.S3.GetObject(env.Ctx, &s3.GetObjectInput{
                   Bucket: aws.String(env.Bucket),
                   Key:    aws.String(key),
               })
               require.NoError(t, err)
               body, err := io.ReadAll(obj.Body)
               require.NoError(t, err)
               sum := blake3.Sum256(body)
               want := "blake3:" + hex.EncodeToString(sum[:])
               require.Equal(t, want, got, "header for key %s does not match BLAKE3 of body", key)
           }
       }
       ```
