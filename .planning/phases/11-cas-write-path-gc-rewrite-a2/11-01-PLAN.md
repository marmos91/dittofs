       ```go
       // FormatCASKey returns the flat S3 object key for a content-addressed block.
       // Format: "cas/{hex[0:2]}/{hex[2:4]}/{hex}". Two-level fanout caps the
       // top-level prefix count at 256 and per-prefix file count predictably.
       // Mirror to ParseCASKey. See BSCAS-01.
       func FormatCASKey(h ContentHash) string {
           hex := hex.EncodeToString(h[:])
           return "cas/" + hex[0:2] + "/" + hex[2:4] + "/" + hex
       }

       // ParseCASKey parses an S3 object key produced by FormatCASKey and
       // returns the embedded ContentHash. Returns ErrCASKeyMalformed wrapped
       // with the offending input on any shape, length, or hex error.
       // Symmetric to FormatCASKey. See BSCAS-01 / D-29.
       func ParseCASKey(key string) (ContentHash, error) {
           const prefix = "cas/"
           if !strings.HasPrefix(key, prefix) {
               return ContentHash{}, fmt.Errorf("%w: missing %q prefix in %q", ErrCASKeyMalformed, prefix, key)
           }
           rest := key[len(prefix):]
           parts := strings.Split(rest, "/")
           if len(parts) != 3 {
               return ContentHash{}, fmt.Errorf("%w: expected 3 segments after prefix in %q", ErrCASKeyMalformed, key)
           }
           shard1, shard2, hexStr := parts[0], parts[1], parts[2]
           if len(shard1) != 2 || len(shard2) != 2 {
               return ContentHash{}, fmt.Errorf("%w: shard segments must be 2 hex chars in %q", ErrCASKeyMalformed, key)
           }
           if len(hexStr) != HashSize*2 {
               return ContentHash{}, fmt.Errorf("%w: hex hash must be %d chars in %q", ErrCASKeyMalformed, HashSize*2, key)
           }
           if hexStr[0:2] != shard1 || hexStr[2:4] != shard2 {
               return ContentHash{}, fmt.Errorf("%w: shard prefix does not match hash in %q", ErrCASKeyMalformed, key)
           }
           raw, err := hex.DecodeString(hexStr)
           if err != nil {
               return ContentHash{}, fmt.Errorf("%w: %v", ErrCASKeyMalformed, err)
           }
           var h ContentHash
           copy(h[:], raw)
           return h, nil
       }
       ```
       Add `"encoding/hex"` and `"strings"` to imports if not already present.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| engine → S3 | DittoFS-owned bytes flow out into a remote object store; metadata header is part of that boundary |
| Hash-to-key derivation | A hash value flows from BLAKE3 output into both an S3 object key and an HTTP header; injection risk if the hash ever came from untrusted input |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-A-01 | Tampering | x-amz-meta-content-hash header | mitigate | Header is set by DittoFS only; ContentHash type is `[32]byte` and CASKey() emits a fixed 7+64 char string — no untrusted bytes can flow through. ParseCASKey enforces strict shape so a malicious key cannot smuggle non-hex chars. |
| T-11-A-02 | Information Disclosure | ParseCASKey error messages | accept | Error messages include the malformed key but the key itself is internal to DittoFS in this code path; not user-controlled. If we ever expose ParseCASKey via a user-facing endpoint, revisit. |
| T-11-A-03 | Spoofing | Legacy zero-valued BlockState rows | mitigate | IsRemote() second arm preserves dual-read for rows with BlockStateDirty(==0) + BlockStoreKey set; the dual-read resolver in plan 03 routes those to legacy paths — no false promotion to CAS. Tested via the dual-read resolver tests in plan 03. |
| T-11-A-04 | Denial of Service | S3 PUT metadata header limit | accept | AWS allows up to 2 KB of user metadata per object; we send ~80 bytes. No risk. |
</threat_model>
