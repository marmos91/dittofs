Detection algorithm (state machine on entry to NewFSStore):
  // baseDir is the shareDir per FSStore convention; sentinel lives at the share root.
  sentinelPath := filepath.Join(baseDir, ".cas-migrated-v1")
  _, err := os.Stat(sentinelPath)
  if err == nil { return constructFSStore(...) }   // sentinel present, normal path
  if !errors.Is(err, fs.ErrNotExist) { return nil, fmt.Errorf("stat sentinel: %w", err) }
  
  // Sentinel missing. Probe for .blk files.
  hasBlk := false
  filepath.WalkDir(baseDir, func(path string, d fs.DirEntry, walkErr error) error {
      if walkErr != nil { return walkErr }
      // Depth guard: limit to 3 directories deep (share / fileID / chunk-or-blk)
      if relDepth(baseDir, path) > 3 { return fs.SkipDir }
      if !d.IsDir() && strings.HasSuffix(d.Name(), ".blk") {
          hasBlk = true
          return fs.SkipAll   // early termination
      }
      return nil
  })
  
  if !hasBlk { return constructFSStore(...) }     // fresh install
  return nil, fmt.Errorf("share %s: %w", baseDir, blockstore.ErrLegacyLayoutDetected)
