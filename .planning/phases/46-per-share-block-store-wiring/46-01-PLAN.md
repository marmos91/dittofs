8. Update `RemoveShare` to close the BlockStore:
```go
func (s *Service) RemoveShare(name string) error {
    s.mu.Lock()
    share, exists := s.registry[name]
    if !exists {
        s.mu.Unlock()
        return fmt.Errorf("share %q not found", name)
    }
    bs := share.BlockStore
    remoteConfigID := "" // extract from share if tracked
    delete(s.registry, name)
    s.mu.Unlock()

    // Close BlockStore outside lock (may block on drain)
    if bs != nil {
        _ = bs.Close(context.Background())
    }

    // Decrement remote store ref count
    if remoteConfigID != "" {
        s.releaseRemoteStore(remoteConfigID)
    }

    s.notifyShareChange()
    return nil
}
```

Add a `remoteConfigID string` field on `Share` to track which remote store config this share uses (needed for ref counting on removal).

9. Add `releaseRemoteStore(configID string)` method:
```go
func (s *Service) releaseRemoteStore(configID string) {
    s.mu.Lock()
    defer s.mu.Unlock()
    sr, ok := s.remoteStores[configID]
    if !ok {
        return
    }
    sr.refCount--
    if sr.refCount <= 0 {
        sr.store.Close()
        delete(s.remoteStores, configID)
    }
}
```
