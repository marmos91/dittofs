Wire Opt 1 (in-memory hash dedup LRU) into the FastCDC chunker emit loop in `pkg/blockstore/local/fs/rollup.go`. On LRU hit: call `FileBlockStore.AddRef`, skip StoreChunk, skip remote Put — block joins the payload's `[]BlockRef` manifest like any other.     chunkBytes := stream[pos : pos+uint64(b)]
    ```go
    h := blake3ContentHash(chunkBytes)
    blockRef := blockstore.BlockRef{Hash: h, Offset: pos, Size: uint32(b)}

    // Phase 19 Opt 1 (D-04): consult the per-FSStore dedup LRU before
    // StoreChunk. On hit, AddRef bumps RefCount on the existing FileBlock
    // row and we skip the local CAS write entirely. The BlockRef append
    // below still happens — D-02 manifest invariant: ComputeObjectID at
    // the end of the loop sees the same BlockRef list with or without LRU.
    skipStoreChunk := false
    if bc.dedupLRU != nil {
        if _, ok := bc.dedupLRU.Get(h); ok {
            err := bc.fileBlockStore.AddRef(ctx, h, payloadID, blockRef)
            switch {
            case err == nil:
                // LRU + metadata both confirm. Skip StoreChunk; STATE-01..03
                // preserved (D-27): AddRef incremented RefCount only.
                skipStoreChunk = true
                bc.log.Debugw("rollup: LRU dedup hit", "hash", h, "payloadID", payloadID, "size", b)
            case errors.Is(err, metadata.ErrUnknownHash):
                // TOCTOU: hash was in LRU but got swept by a concurrent
                // engine.Delete cascade. Fall through to the standard
                // StoreChunk path; the post-StoreChunk LRU repopulate below
                // re-seeds with the freshly-written row.
                bc.log.Debugw("rollup: LRU stale (ErrUnknownHash); falling back to StoreChunk", "hash", h)
            default:
                return fmt.Errorf("rollup: AddRef: %w", err)
            }
        }
    }

    if !skipStoreChunk {
        if err := bc.StoreChunk(ctx, h, chunkBytes); err != nil {
            return fmt.Errorf("rollup: StoreChunk: %w", err)
        }
        // Phase 19 Opt 1: first-write seeds the LRU for subsequent
        // idempotent rewrites.
        if bc.dedupLRU != nil {
            bc.dedupLRU.Put(h, payloadID)
        }
    }

    blocks = append(blocks, blockRef)
    pos += uint64(b)
    ```
